### 测试用例 9: API-1 会话按 agent_name 隔离存储

### 验证

```run-code
# 验证所有智能体的会话 ID 不重复
const agents = ['test001', 'skill-test-pdf', 'skill-test-doc'];
const allIds = [];
for (const agent of agents) {
  const resp = await fetch(`http://localhost:20881/api/agents/${agent}/conversations`);
  const data = await resp.json();
  data.conversations.forEach(c => allIds.push(c.id));
}
const uniqueIds = [...new Set(allIds)];
console.assert(allIds.length === uniqueIds.length, 'All conversation IDs should be unique');
console.log(`Total: ${allIds.length}, Unique: ${uniqueIds.length}`);
```

### 验证

```run-code
# 创建新会话、保存消息、读取验证
const createResp = await fetch('http://localhost:20881/api/agents/test001/conversations', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ title: 'API测试会话' })

const createData = await createResp.json();
console.assert(createData.id !== undefined, 'Should have conversation ID');

const saveResp = await fetch(`http://localhost:20881/api/agents/test001/conversations/${createData.id}/save`, {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    messages: [
      { id: 'test-msg-1', role: 'user', content: 'API测试消息' },
      { id: 'test-msg-2', role: 'assistant', content: 'API测试回复' }
    ]
  })
});
const saveData = await saveResp.json();
console.assert(saveData.success === true, 'Save should succeed');

const getResp = await fetch(`http://localhost:20881/api/agents/test001/conversations/${createData.id}`);
const getData = await getResp.json();
console.assert(getData.messages.length === 2, 'Should have 2 messages');
console.log('API-3 PASS: Conversation saved and retrieved');
```
