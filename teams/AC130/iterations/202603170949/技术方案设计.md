### 2.3 文档处理器 (document_processor.py)

```python
from unstructured.partition.pdf import partition_pdf
from unstructured.partition.docx import partition_docx
from unstructured.chunking.title import chunk_by_title

class DocumentProcessor:
    """文档解析与智能分块"""

    def __init__(self, chunk_size: int = 1000, chunk_overlap: int = 100):
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap

    def process(self, file_path: str, file_type: str) -> List[dict]:
        """
        处理文档，返回分块列表

        Returns:
            List[dict]: [{"content": str, "metadata": dict}, ...]
        """
        # 1. 解析文档
        if file_type == "pdf":
            elements = partition_pdf(file_path)
        elif file_type == "docx":
            elements = partition_docx(file_path)
        else:
            raise ValueError(f"Unsupported file type: {file_type}")

        # 2. 智能分块 (by_title 策略)
        chunks = chunk_by_title(
            elements,
            max_characters=self.chunk_size,
            overlap=self.chunk_overlap
        )

        # 3. 转换为标准格式
        result = []
        for i, chunk in enumerate(chunks):
            result.append({
                "content": str(chunk),
                "metadata": {
                    "chunk_index": i,
                    "page": getattr(chunk.metadata, 'page_number', None)
                }
            })

        return result
```

### 2.4 向量化服务 (embedder.py)

```python
from zhipuai import ZhipuAI
from typing import List
import os

class Embedder:
    """向量化服务（使用 ZhipuAI Embedding API）"""

    def __init__(self):
        self.client = ZhipuAI(api_key=os.getenv("ZHIPU_API_KEY"))
        self.model = "embedding-3"  # 智谱 Embedding 模型

    def embed_documents(self, texts: List[str]) -> List[List[float]]:
        """批量向量化文本"""
        response = self.client.embeddings.create(
            model=self.model,
            input=texts
        )
        return [item.embedding for item in response.data]

    def embed_query(self, query: str) -> List[float]:
        """向量化查询"""
        response = self.client.embeddings.create(
            model=self.model,
            input=[query]
        )
        return response.data[0].embedding
```

### 2.5 检索服务 (retriever.py)

```python
import chromadb
from chromadb.config import Settings
from typing import List, Optional
from .embedder import Embedder
from .models import SearchResult, SearchResponse

class Retriever:
    """向量检索服务"""

    def __init__(self, persist_directory: str = "data/chromadb"):
        self.client = chromadb.PersistentClient(path=persist_directory)
        self.embedder = Embedder()

    def get_or_create_collection(self, kb_id: str):
        """获取或创建 Collection"""
        return self.client.get_or_create_collection(
            name=f"kb_{kb_id}",
            metadata={"hnsw:space": "cosine"}
        )

    def add_documents(
        self,
        kb_id: str,
        chunks: List[dict],
        doc_id: str,
        filename: str
    ):
        """添加文档块到向量库"""
        collection = self.get_or_create_collection(kb_id)

        texts = [chunk["content"] for chunk in chunks]
        embeddings = self.embedder.embed_documents(texts)

        ids = [f"{doc_id}_{i}" for i in range(len(chunks))]
        metadatas = [
            {
                "doc_id": doc_id,
                "filename": filename,
                "page": chunk["metadata"].get("page"),
                "chunk_index": chunk["metadata"]["chunk_index"]
            }
            for chunk in chunks
        ]

        collection.add(
            ids=ids,
            embeddings=embeddings,
            documents=texts,
            metadatas=metadatas
        )

    def search(
        self,
        kb_id: str,
        query: str,
        top_k: int = 5
    ) -> SearchResponse:
        """搜索知识库"""
        collection = self.get_or_create_collection(kb_id)
        query_embedding = self.embedder.embed_query(query)

        results = collection.query(
            query_embeddings=[query_embedding],
            n_results=top_k,
            include=["documents", "metadatas", "distances"]
        )

        search_results = []
        for i in range(len(results["ids"][0])):
            search_results.append(SearchResult(
                content=results["documents"][0][i],
                filename=results["metadatas"][0][i]["filename"],
                page=results["metadatas"][0][i].get("page"),
                chunk_index=results["metadatas"][0][i]["chunk_index"],
                score=1 - results["distances"][0][i]  # 转换为相似度
            ))

        return SearchResponse(
            results=search_results,
            query=query,
            kb_name=kb_id,
            total=len(search_results)
        )

    def delete_document(self, kb_id: str, doc_id: str):
        """删除文档的所有块"""
        collection = self.get_or_create_collection(kb_id)
        # 获取所有该文档的 chunk IDs
        all_items = collection.get()
        ids_to_delete = [
            id for id, meta in zip(all_items["ids"], all_items["metadatas"])
            if meta["doc_id"] == doc_id
        ]
        if ids_to_delete:
            collection.delete(ids=ids_to_delete)

    def delete_collection(self, kb_id: str):
        """删除整个 Collection"""
        try:
            self.client.delete_collection(name=f"kb_{kb_id}")
        except:
            pass
```

### 2.6 知识库管理器 (knowledge_base_manager.py)

```python
import json
import os
from typing import List, Optional
from pathlib import Path
from .models import KnowledgeBase, KnowledgeBaseCreate, KnowledgeBaseUpdate, Document
from .document_processor import DocumentProcessor
from .retriever import Retriever

class KnowledgeBaseManager:
    """知识库管理核心类"""

    def __init__(self, data_dir: str = "data"):
        self.data_dir = Path(data_dir)
        self.kb_file = self.data_dir / "knowledge_bases.json"
        self.docs_dir = self.data_dir / "kb_documents"
        self.files_dir = self.data_dir / "kb_files"

        # 确保目录存在
        self.docs_dir.mkdir(parents=True, exist_ok=True)
        self.files_dir.mkdir(parents=True, exist_ok=True)

        self.processor = DocumentProcessor()
        self.retriever = Retriever()
        self._load_knowledge_bases()

    def _load_knowledge_bases(self):
        """加载知识库数据"""
        if self.kb_file.exists():
            with open(self.kb_file, "r", encoding="utf-8") as f:
                self.knowledge_bases = {
                    kb["id"]: KnowledgeBase(**kb)
                    for kb in json.load(f)
                }
        else:
            self.knowledge_bases = {}

    def _save_knowledge_bases(self):
        """保存知识库数据"""
        with open(self.kb_file, "w", encoding="utf-8") as f:
            json.dump(
                [kb.model_dump() for kb in self.knowledge_bases.values()],
                f,
                ensure_ascii=False,
                indent=2,
                default=str
            )

    def create(self, req: KnowledgeBaseCreate) -> KnowledgeBase:
        """创建知识库"""
        kb = KnowledgeBase(name=req.name, description=req.description)
        self.knowledge_bases[kb.id] = kb
        self._save_knowledge_bases()

        # 创建文档索引文件
        kb_docs_file = self.docs_dir / f"{kb.id}.json"
        with open(kb_docs_file, "w", encoding="utf-8") as f:
            json.dump([], f)

        return kb

    def get(self, kb_id: str) -> Optional[KnowledgeBase]:
        """获取知识库"""
        return self.knowledge_bases.get(kb_id)

    def list(self) -> List[KnowledgeBase]:
        """列出所有知识库"""
        return list(self.knowledge_bases.values())

    def update(self, kb_id: str, req: KnowledgeBaseUpdate) -> Optional[KnowledgeBase]:
        """更新知识库"""
        kb = self.knowledge_bases.get(kb_id)
        if not kb:
            return None

        if req.name is not None:
            kb.name = req.name
        if req.description is not None:
            kb.description = req.description

        kb.updated_at = datetime.now()
        self._save_knowledge_bases()
        return kb

    def delete(self, kb_id: str) -> bool:
        """删除知识库（级联删除文档和向量）"""
        if kb_id not in self.knowledge_bases:
            return False

        # 1. 删除向量库
        self.retriever.delete_collection(kb_id)

        # 2. 删除文件
        kb_files_dir = self.files_dir / kb_id
        if kb_files_dir.exists():
            import shutil
            shutil.rmtree(kb_files_dir)

        # 3. 删除文档索引
        kb_docs_file = self.docs_dir / f"{kb_id}.json"
        if kb_docs_file.exists():
            kb_docs_file.unlink()

        # 4. 删除知识库记录
        del self.knowledge_bases[kb_id]
        self._save_knowledge_bases()

        return True

    def add_document(
        self,
        kb_id: str,
        file_path: str,
        filename: str,
        file_size: int
    ) -> Document:
        """添加文档到知识库"""
        kb = self.knowledge_bases.get(kb_id)
        if not kb:
            raise ValueError(f"Knowledge base not found: {kb_id}")

        # 创建文档记录
        doc = Document(
            kb_id=kb_id,
            filename=filename,
            file_path=file_path,
            file_size=file_size,
            status="processing"
        )

        # 保存文档记录
        kb_docs_file = self.docs_dir / f"{kb_id}.json"
        with open(kb_docs_file, "r", encoding="utf-8") as f:
            docs = json.load(f)
        docs.append(doc.model_dump())
        with open(kb_docs_file, "w", encoding="utf-8") as f:
            json.dump(docs, f, ensure_ascii=False, indent=2, default=str)

        try:
            # 解析文档
            file_type = filename.split(".")[-1].lower()
            chunks = self.processor.process(file_path, file_type)

            # 向量化并存储
            self.retriever.add_documents(
                kb_id=kb_id,
                chunks=chunks,
                doc_id=doc.id,
                filename=filename
            )

            # 更新文档状态
            doc.status = "indexed"
            doc.chunk_count = len(chunks)

        except Exception as e:
            doc.status = "error"
            doc.error_message = str(e)

        # 更新文档记录
        docs = [d if d["id"] != doc.id else doc.model_dump() for d in docs]
        with open(kb_docs_file, "w", encoding="utf-8") as f:
            json.dump(docs, f, ensure_ascii=False, indent=2, default=str)

        # 更新知识库文档计数
        kb.document_count = len([d for d in docs if d["status"] == "indexed"])
        self._save_knowledge_bases()

        return doc

    def list_documents(self, kb_id: str) -> List[Document]:
        """列出知识库的所有文档"""
        kb_docs_file = self.docs_dir / f"{kb_id}.json"
        if not kb_docs_file.exists():
            return []

        with open(kb_docs_file, "r", encoding="utf-8") as f:
            docs = json.load(f)

        return [Document(**d) for d in docs]

    def delete_document(self, kb_id: str, doc_id: str) -> bool:
        """删除文档"""
        kb_docs_file = self.docs_dir / f"{kb_id}.json"
        if not kb_docs_file.exists():
            return False

        with open(kb_docs_file, "r", encoding="utf-8") as f:
            docs = json.load(f)

        # 找到文档
        doc = next((d for d in docs if d["id"] == doc_id), None)
        if not doc:
            return False

        # 删除文件
        if os.path.exists(doc["file_path"]):
            os.remove(doc["file_path"])

        # 删除向量
        self.retriever.delete_document(kb_id, doc_id)

        # 更新文档列表
        docs = [d for d in docs if d["id"] != doc_id]
        with open(kb_docs_file, "w", encoding="utf-8") as f:
            json.dump(docs, f, ensure_ascii=False, indent=2, default=str)

        # 更新知识库文档计数
        kb = self.knowledge_bases.get(kb_id)
        if kb:
            kb.document_count = len([d for d in docs if d["status"] == "indexed"])
            self._save_knowledge_bases()

        return True
```

                            }

                        "required": ["query"]
                    }
                }
            })
    return tools

# 新增方法：处理知识库工具调用
async def _handle_kb_tool_call(self, tool_name: str, args: dict) -> str:
    """处理知识库工具调用"""
    # 解析 kb_id from tool_name (query_kb_xxxxxxxx)
    kb_id_prefix = tool_name.replace("query_kb_", "")
    kb = next((kb for kb in self.kb_manager.list()
               if kb.id.startswith(kb_id_prefix)), None)

    if not kb:
        return "知识库不存在"

    # 检索
    results = self.retriever.search(kb.id, args["query"])

    # 格式化结果
    context_parts = []
    for r in results.results:
        source = f"《{r.filename}》"
        if r.page:
            source += f" 第{r.page}页"
        context_parts.append(f"【{source}】\n{r.content}")

    return f"检索到 {results.total} 条相关内容：\n\n" + "\n\n---\n\n".join(context_parts)

### 3.3 智能体配置选择器 (KnowledgeBaseSelector.tsx)

```tsx
import { useEffect, useState } from "react";
import { Checkbox } from "@/components/ui/checkbox";
import { Label } from "@/components/ui/label";
import { kbApi, KnowledgeBase } from "@/lib/knowledgeBaseApi";

interface KnowledgeBaseSelectorProps {

  onChange: (kbIds: string[]) => void;

export function KnowledgeBaseSelector({
  agentName,
  selectedKbIds,
  onChange,
}: KnowledgeBaseSelectorProps) {

      onChange(selectedKbIds.filter((id) => id !== kbId));
    }
  };

  return (
    <div className="space-y-3">
      <h4 className="font-medium text-sm">RAG 知识库</h4>
      <p className="text-xs text-gray-500">
        勾选的知识库将作为工具挂载到智能体，描述信息将帮助智能体判断何时调用
      </p>

      {knowledgeBases.length === 0 ? (
        <p className="text-sm text-gray-400">暂无可用知识库</p>
      ) : (
        <div className="space-y-2">
          {knowledgeBases.map((kb) => (
            <div key={kb.id} className="flex items-start space-x-2 p-2 rounded hover:bg-gray-50">
              <Checkbox
                id={`kb-${kb.id}`}
                checked={selectedKbIds.includes(kb.id)}
                onCheckedChange={(checked) => handleToggle(kb.id, checked as boolean)}
              />
              <div className="flex-1">
                <Label htmlFor={`kb-${kb.id}`} className="font-medium cursor-pointer">
                  {kb.name}
                </Label>
                <p className="text-xs text-gray-500 mt-0.5">{kb.description}</p>
              </div>
            </div>
          ))}
        </div>
      )}
    </div>
  );
}
```
