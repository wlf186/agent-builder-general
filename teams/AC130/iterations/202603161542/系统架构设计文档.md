### 2.2 DocumentProcessor（文档处理器）

**职责**: 文档解析、文本分块、向量化

**文件位置**: `src/document_processor.py`

```python
class DocumentProcessor:
    """文档处理流水线"""

    def __init__(self, embedding_service, chunk_size: int = 512, chunk_overlap: int = 50):
        self.embedding_service = embedding_service
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap

    async def process_document(self, file_path: str, file_type: str) -> ProcessResult:
        """
        处理文档的完整流程

        Args:
            file_path: 文件路径
            file_type: 文件类型 (pdf, docx)

        Returns:
            ProcessResult: 包含 chunks 和 embeddings
        """
        # 1. 解析文档
        text = await self._parse_document(file_path, file_type)

        # 2. 分块
        chunks = self._split_text(text)

        # 3. 向量化
        embeddings = await self._generate_embeddings(chunks)

        return ProcessResult(chunks=chunks, embeddings=embeddings)

    async def _parse_document(self, file_path: str, file_type: str) -> str:
        """解析文档，提取文本"""
        if file_type == "pdf":
            return await self._parse_pdf(file_path)
        elif file_type == "docx":
            return await self._parse_docx(file_path)
        else:
            raise ValueError(f"Unsupported file type: {file_type}")

    async def _parse_pdf(self, file_path: str) -> str:
        """解析 PDF 文件"""
        # 优先使用 Unstructured
        try:
            from unstructured.partition.pdf import partition_pdf
            elements = partition_pdf(filename=file_path)
            return "\n\n".join([str(el) for el in elements])
        except Exception as e:
            # 回退到 PyMuPDF4LLM
            import fitz  # PyMuPDF
            doc = fitz.open(file_path)
            text = ""
            for page in doc:
                text += page.get_text()
            return text

    async def _parse_docx(self, file_path: str) -> str:
        """解析 DOCX 文件"""
        from unstructured.partition.docx import partition_docx
        elements = partition_docx(filename=file_path)
        return "\n\n".join([str(el) for el in elements])

    def _split_text(self, text: str) -> List[TextChunk]:
        """文本分块"""
        from langchain.text_splitter import RecursiveCharacterTextSplitter

        splitter = RecursiveCharacterTextSplitter(
            chunk_size=self.chunk_size,
            chunk_overlap=self.chunk_overlap,
            separators=["\n\n", "\n", "。", "！", "？", "；", " ", ""]
        )

        chunks = splitter.split_text(text)
        return [TextChunk(content=chunk, index=i) for i, chunk in enumerate(chunks)]

    async def _generate_embeddings(self, chunks: List[TextChunk]) -> List[List[float]]:
        """生成向量嵌入"""
        texts = [chunk.content for chunk in chunks]
        embeddings = await self.embedding_service.embed_documents(texts)
        return embeddings
```

### 2.3 RAGEngine（RAG 引擎）

**职责**: 向量检索、结果排序、与 Agent 引擎集成

**文件位置**: `src/rag_engine.py`

```python
from chromadb import Client, Config
from typing import List, Optional
import chromadb

class RAGEngine:
    """RAG 检索引擎"""

    def __init__(self, persist_directory: str = "data/chroma"):
        self.client = chromadb.PersistentClient(path=persist_directory)
        self._collections = {}  # kb_id -> Collection 缓存

    def get_or_create_collection(self, kb_id: str):
        """获取或创建知识库对应的 Collection"""
        if kb_id not in self._collections:
            self._collections[kb_id] = self.client.get_or_create_collection(
                name=f"kb_{kb_id}",
                metadata={"hnsw:space": "cosine"}
            )
        return self._collections[kb_id]

    def add_chunks(self, kb_id: str, chunks: List[TextChunk],
                   embeddings: List[List[float]], doc_id: str):
        """添加文档块到向量库"""
        collection = self.get_or_create_collection(kb_id)

        ids = [f"{doc_id}_{i}" for i in range(len(chunks))]
        metadatas = [
            {
                "doc_id": doc_id,
                "chunk_index": chunk.index,
                "filename": chunk.filename
            }
            for chunk in chunks
        ]

        collection.add(
            ids=ids,
            embeddings=embeddings,
            documents=[chunk.content for chunk in chunks],
            metadatas=metadatas
        )

    def delete_document(self, kb_id: str, doc_id: str):
        """删除文档的所有块"""
        collection = self.get_or_create_collection(kb_id)
        # 查询该文档的所有 chunk IDs
        results = collection.get(
            where={"doc_id": doc_id}
        )
        if results["ids"]:
            collection.delete(ids=results["ids"])

    def delete_collection(self, kb_id: str):
        """删除整个知识库的 Collection"""
        try:
            self.client.delete_collection(name=f"kb_{kb_id}")
            if kb_id in self._collections:
                del self._collections[kb_id]
        except Exception:
            pass

    async def search(self, kb_id: str, query: str,
                     query_embedding: List[float],
                     n_results: int = 5) -> List[SearchResult]:
        """
        检索相关文档块

        Args:
            kb_id: 知识库 ID
            query: 查询文本
            query_embedding: 查询向量
            n_results: 返回结果数量

        Returns:
            SearchResult 列表
        """
        collection = self.get_or_create_collection(kb_id)

        results = collection.query(
            query_embeddings=[query_embedding],
            n_results=n_results,
            include=["documents", "metadatas", "distances"]
        )

        search_results = []
        for i in range(len(results["ids"][0])):
            search_results.append(SearchResult(
                chunk_id=results["ids"][0][i],
                content=results["documents"][0][i],
                metadata=results["metadatas"][0][i],
                score=1 - results["distances"][0][i]  # 转换为相似度
            ))

        return search_results
```

### 2.4 Agent 集成（AgentEngine 改造）

**职责**: 在 Agent 引擎中动态注入 RAG 工具

**文件位置**: `src/agent_engine.py` (修改)

```python
class AgentEngine:
    # ... 现有代码 ...

    def _build_rag_tools(self) -> List[Dict]:
        """
        根据当前 Agent 挂载的知识库，构建 RAG 工具定义

        Returns:
            工具定义列表
        """
        kb_list = self.kb_manager.get_agent_kbs(self.agent_name)

        if not kb_list:
            return []

        # 构建工具描述
        kb_descriptions = "\n".join([
            f"- {kb.name}: {kb.description}"
            for kb in kb_list
        ])

        tool = {
            "type": "function",
            "function": {
                "name": "search_knowledge_base",
                "description": f"""搜索指定的知识库以获取相关文档内容。

可用的知识库:
{kb_descriptions}

仅在与上述知识库描述相关的查询时使用此工具。""",
                "parameters": {
                    "type": "object",
                    "properties": {
                        "kb_name": {
                            "type": "string",
                            "description": f"知识库名称，可选值: {[kb.name for kb in kb_list]}"
                        },
                        "query": {
                            "type": "string",
                            "description": "搜索查询语句"
                        }
                    },
                    "required": ["kb_name", "query"]
                }
            }
        }

        return [tool]

    async def _execute_rag_tool(self, kb_name: str, query: str) -> str:
        """
        执行 RAG 检索工具

        Args:
            kb_name: 知识库名称
            query: 查询语句

        Returns:
            检索结果文本
        """
        # 1. 获取知识库
        kb = self.kb_manager.get_kb_by_name(kb_name)
        if not kb:
            return f"错误: 未找到知识库 '{kb_name}'"

        # 2. 生成查询向量
        query_embedding = await self.embedding_service.embed_query(query)

        # 3. 检索
        results = await self.rag_engine.search(
            kb_id=kb.id,
            query=query,
            query_embedding=query_embedding,
            n_results=5
        )

        if not results:
            return f"在知识库 '{kb_name}' 中未找到相关内容。"

        # 4. 格式化结果
        context_parts = []
        sources = []

        for i, result in enumerate(results, 1):
            context_parts.append(f"[{i}] {result.content}")
            sources.append(result.metadata.get("filename", "未知文档"))

        context = "\n\n".join(context_parts)
        sources_str = ", ".join(set(sources))

        return f"""找到以下相关内容:

{context}

---
📚 参考来源: {sources_str}"""
```

### 5.3 Agent 配置页集成

在现有智能体配置页面添加知识库选择组件：

```tsx
// frontend/src/components/AgentKnowledgeBaseSelector.tsx
interface Props {
  agentName: string;
  selectedKbIds: string[];
  onChange: (kbIds: string[]) => void;
}

export function AgentKnowledgeBaseSelector({ agentName, selectedKbIds, onChange }: Props) {
  const [knowledgeBases, setKnowledgeBases] = useState([]);

  useEffect(() => {
    fetch('/api/knowledge-bases')
      .then(res => res.json())
      .then(data => setKnowledgeBases(data.knowledge_bases));
  }, []);

  return (
    <div className="space-y-2">
      <h3 className="font-semibold">RAG 知识库</h3>
      {knowledgeBases.map(kb => (
        <label key={kb.id} className="flex items-center gap-2">
          <input
            type="checkbox"
            checked={selectedKbIds.includes(kb.id)}
            onChange={(e) => {
              if (e.target.checked) {
                onChange([...selectedKbIds, kb.id]);
              } else {
                onChange(selectedKbIds.filter(id => id !== kb.id));
              }
            }}
          />
          <span>{kb.name}</span>
          <span className="text-gray-500 text-sm">- {kb.description}</span>
        </label>
      ))}
    </div>
  );
}
```
