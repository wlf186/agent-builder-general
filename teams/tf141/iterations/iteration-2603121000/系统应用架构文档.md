```python
# src/environment_manager.py 新增方法

async def install_skill_dependencies(
    self,
    agent_name: str,
    skill_path: Path,
    skill_name: str
) -> Tuple[bool, str, List[str]]:
    """
    为 Skill 安装依赖

    Args:
        agent_name: Agent 名称
        skill_path: Skill 目录路径
        skill_name: Skill 名称

    Returns:
        (success, message, installed_packages)
    """
    requirements_path = skill_path / "scripts" / "requirements.txt"

    if not requirements_path.exists():
        return True, "No requirements.txt found", []

    # 读取依赖列表
    with open(requirements_path, 'r') as f:
        packages = [
            line.strip()
            for line in f
            if line.strip() and not line.startswith('#')
        ]

    if not packages:
        return True, "No packages to install", []

    # 检查是否已安装
    environment = await self.get_environment_status(agent_name)
    if environment and skill_name in environment.installed_dependencies:
        return True, "Dependencies already installed", []

    # 安装依赖
    success, message = await self.install_packages(agent_name, packages)

    if success:
        # 更新已安装依赖记录
        if environment:
            environment.installed_dependencies[skill_name] = packages
            environment.updated_at = datetime.now().isoformat()
            self._save_metadata(environment)

    return success, message, packages if success else []
```

```python
# src/models.py 新增

class FileContext(BaseModel):
    """文件上下文，用于 Agent 感知上传的文件"""
    file_ids: List[str] = []
    file_infos: List[Dict[str, Any]] = []  # 简化的文件信息列表

    def to_context_string(self) -> str:
        """生成给 LLM 的上下文字符串"""
        if not self.file_ids:
            return ""

        lines = ["当前对话中用户上传了以下文件："]
        for info in self.file_infos:
            lines.append(f"  - 文件ID: {info.get('file_id')}")
            lines.append(f"    文件名: {info.get('filename')}")
            lines.append(f"    类型: {info.get('mime_type')}")
            lines.append(f"    大小: {info.get('file_size', 0) / 1024:.1f} KB")

        lines.append("\n调用 execute_skill 工具时，请使用上述 file_id。")
        return "\n".join(lines)
```
